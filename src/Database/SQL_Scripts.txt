CREATE TABLE IF NOT EXISTS tb_1to1_SavedEmails (
    IDEmail INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    DataCad DATETIME NOT NULL DEFAULT NOW(),
	Email_From VARCHAR(300) NOT NULL,
	Email_To VARCHAR(300) NOT NULL,
	Email_Subject VARCHAR(300) NOT NULL,
	Email_Body MEDIUMTEXT NOT NULL,
	Email_TimeStamp DATETIME NOT NULL,
	Email_HasAttach INT NOT NULL DEFAULT 0
    );
    

CREATE TABLE IF NOT EXISTS tb_AppCal_Meetings (
	IDEmail INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
	DataCad DATETIME NOT NULL DEFAULT NOW(),
	Email_From VARCHAR(300) NOT NULL,
	Email_To VARCHAR(300) NOT NULL,
	Email_Subject VARCHAR(300) NOT NULL,
	Email_Body MEDIUMTEXT NOT NULL,
	Email_TimeStamp DATETIME NOT NULL,
	Email_HasAttach INT NOT NULL DEFAULT 0
);
